\begin{frame}{考案に至った理由}
  \begin{itemize}
    \item 公表ベースライン：\textbf{PBL 4.38 m / CLA 8.07 m / WCL 3.57 m}（往路平均誤差）\cite{tanaka2023tut}，目標は\textbf{平均誤差 2 m 未満}
    \item WCL\cite{blumenthal2007wcl}（centroid 系\cite{bulusu2000gpsless} の重み付き拡張）の構造的問題：推定位置が観測 AP 群の\textbf{凸包内に閉じ込められ}，境界に近い地点ほど内側へ引き寄せられる\cite{ji2012vap}
    \item この境界バイアスは重みの付け方（power-domain 化・top-$L$・blacklist 等）を工夫しても\textbf{原理的に解消できない}
    \item 着目：廊下という伝搬環境は\textbf{L 字の 1 次元多様体}であり，壁越しに近い 2 点も Euclid 距離では近接に見えてしまう（\textbf{偽相関}）
    \item $\Rightarrow$ 廊下の\textbf{弧長座標}上で AP ごとの電波伝搬を直接モデル化する \textbf{1D Gaussian Process radio map（gp\_corridor）}を考案
  \end{itemize}
\end{frame}

\begin{frame}{原理①：モデル定義式}
  \begin{itemize}
    \item 廊下中心線の弧長座標 $s\in[0,S]$ 上で，鍵 (AP, band) ごとに RSSI を 1 次元 GP としてモデル化：$r_{a,b}(s)\sim\mathcal{GP}(m(s),k(s,s'))$
    \item カーネルは Matérn-3/2（AP ごとに長さ尺度 $\ell$ を独立に学習）：
    \[
      k(s,s')=\sigma_f^2\Bigl(1+\tfrac{\sqrt3\,|s-s'|}{\ell}\Bigr)\exp\Bigl(-\tfrac{\sqrt3\,|s-s'|}{\ell}\Bigr)
    \]
    \item クエリ地点の位置対数尤度（GP 事後平均 $\mu(s)$・分散 $v(s)$ を用いる）：
    \[
      \log p(s\mid r)=-\frac12\sum_{a,b}\Bigl[\frac{(r-\mu(s))^2}{v(s)+\sigma^2}+\log\bigl(v(s)+\sigma^2\bigr)\Bigr]
    \]
    \item 階層構成：セグメント分類（C/C2/C3）$\to$ セグメント内で 1D 尤度推定\cite{ferris2007gpslam}
  \end{itemize}
\end{frame}

\begin{frame}{原理②：仕様と処理の流れ}
  \small
  \begin{enumerate}
    \item 前処理：測量座標 $(x,y)$ を廊下弧長 $s$ へ変換
    \item 鍵 (AP, band) ごとに GP を fit：train の周辺尤度（LML）グリッド探索で $\ell\in\{2,\dots,32\}$ m，$\sigma_f\in\{2,5,10\}$ dB，$\sigma_n\in\{1,2,3\}$ dB を選択
    \item 102 鍵中，\textbf{検出 5 地点以上}の \textbf{82 鍵}のみモデル化（希薄な鍵は除外）
    \item クエリ時：セグメント分類 $\to$ 該当セグメント内で鍵ごとの対数尤度を累積 $\to$ $\arg\max_s$ で弧長推定 $\to$ $(x,y)$ へ復元
  \end{enumerate}
  \vspace{4pt}
  \begin{itemize}\small
    \item \textbf{契約}：\texttt{fit} には train のスキャンと train 地点座標のみを渡す（test 情報は一切不使用）
  \end{itemize}
\end{frame}

\begin{frame}{立証過程：この式が正しいことをどう確かめたか}
  \begin{itemize}
    \item \textbf{評価設計}：Protocol A（2 fold，同一 59 地点の再認識）と LOLO（59 fold，train=往路から保持地点を除外，test=\emph{復路}の当該地点）を使い分け $\Rightarrow$ LOLO を\textbf{空間+方向汎化の決定的テスト}と位置付け
    \item \textbf{リーク対策}：分割 iterator による構造的除外（held-out 地点を \texttt{fit} に渡さない）$+$ spy テスト \texttt{test\_iter\_lolo\_leakage\_contract\_spy} で検証
    \item \textbf{統計的立証}：地点別誤差の paired bootstrap（$B{=}1000$，seed$=$0）で 95\% CI を算出し，WCL との差の有意性を判定\cite{hara2010statloc}
  \end{itemize}
\end{frame}

\begin{frame}{有効性③：なぜ効くのか（機構）}
  \begin{itemize}
    \item 廊下が\textbf{1 次元多様体}であるという事前情報を明示的に利用 $\Rightarrow$ 2D $(x,y)$ でなく弧長 $s$ で回帰し\textbf{有効自由度を削減}
    \item AP (band) 毎に長さ尺度 $\ell$ を\textbf{独立に学習} $\Rightarrow$ 空間的平滑性の異なる AP を個別に扱える
    \item セグメント分類（C/C2/C3, train accuracy 1.00）が\textbf{壁越しの偽相関を遮断}
    \item LOLO（0.72）$<$ Protocol A（1.01, fold 平均）の逆転：Protocol A では方向依存オフセットへの過適合という\textbf{仮説}（断定はしない）
    \item この逆転はリークでは説明できない：spy テストにより構造的除外を確認済み
  \end{itemize}
\end{frame}

\begin{frame}{比較：なぜ gp\_corridor が主役なのか}
  \begin{itemize}
    \item \textbf{指紋 (FP) 系 4 手法}（wknn/centered/studentt/rank）：WCL 3.51 m を大幅改善するが\textbf{2 m 級止まり}（目標未達）
    \item \textbf{Tier 3：WCL 改良 7 種}（power-domain・linpower・top-$L$・blacklist・varweight・multiband・廊下射影）：\textbf{凸包制約を保持}したままの重み調整のため最良でも 3.31 m 止まり
    \item \textbf{Tier 4 追補 7 手法}（Mahalanobis/GP-augmented/joint\_fp/Fisher/PLS/ordinal/wcl\_residual）：\textbf{vs gp\_corridor を事前指定}した主要比較で\textbf{全て有意に劣後}
      \begin{itemize}
        \item 最良は mahalanobis\_wknn 1.77 m（$\Delta{=}+1.04$，95\% CI $[0.69,1.38]$）
      \end{itemize}
    \item $\Rightarrow$ 19 手法中，目標 2 m 未満を達成したのは \textbf{gp\_corridor のみ}
  \end{itemize}
\end{frame}

\begin{frame}{付録C：代替手法の定義式}
  \small
  \textbf{studentt\_fp}（Student-$t$ 尤度 $+$ Bernoulli 検出項 $+$ 共通オフセット補正，$\nu{=}3$）\cite{youssef2005horus}
  \[
    \log p(s\mid q)=\sum_{a,b}\Bigl[\beta_{a,b}\,D_{a,b}\,E_{s,a,b}\,\log t_\nu\bigl(r;\,\mu(s)+\hat\delta_q,\,\sigma(s)\bigr)+\log p(D_{a,b}\mid s)\Bigr]
  \]
  \[
    \hat\delta_q=\operatorname*{median}_{(a,b):\,D_{a,b}E_{s,a,b}=1}\bigl[r_{q,a,b}-\mu_{a,b}(s)\bigr]
  \]
  \vspace{4pt}
  \textbf{centered\_fp}（scan 共通オフセット除去，$\lambda{=}0.25$）
  \[
    \tilde r_{s,a}=r_{s,a}-\operatorname*{median}_{j\in O_s} r_{s,j}
  \]
  \[
    d^2=\lambda\sum_a w_a(r-\mu)^2+(1-\lambda)\sum_a \tilde w_a(\tilde r-\tilde\mu)^2
  \]
\end{frame}
